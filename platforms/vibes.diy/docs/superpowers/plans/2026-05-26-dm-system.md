# DM System Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement a 1:1 direct message system where threads live at `_d.userHandleA.userHandleB/dm` in the existing AppDocuments infrastructure, with unread badges in the pill and sidebar.

**Architecture:** DM threads are fake vibes in a reserved `_d.*` namespace. The existing putDoc/queryDocs/subscribeDocs/deleteDoc handlers serve DMs after a new ACL branch that checks participant membership rather than vibe membership. Two new tables (DirectChannelIndex, DirectChannelReads) enable fast thread listing and unread counts. Three new wire types (listDmThreads, markDmRead, req-open-dm-thread) round out the API.

**Tech Stack:** Drizzle ORM (SQLite + Postgres), ArkType, Cloudflare Workers, React, React Router v7, Clerk auth, vitest

**Spec:** `docs/superpowers/specs/2026-05-26-dm-system-design.md`  
**Issue:** https://github.com/VibesDIY/vibes.diy/issues/1947

```typescript
export const DIRECT_CHANNEL_PREFIX = "_d.";

export function isDirectChannel(userHandle: string): boolean {
  return userHandle.startsWith(DIRECT_CHANNEL_PREFIX);
}

export function directChannelUserSlug(a: string, b: string): string {
  const [p, q] = [a, b].sort();
  return `${DIRECT_CHANNEL_PREFIX}${p}.${q}`;
}

export function directChannelParticipants(channelUserSlug: string): [string, string] | undefined {
  if (!channelUserSlug.startsWith(DIRECT_CHANNEL_PREFIX)) return undefined;
  const rest = channelUserSlug.slice(DIRECT_CHANNEL_PREFIX.length);
  const parts = rest.split(".");
  if (parts.length !== 2) return undefined;
  return [parts[0], parts[1]];
}
```

```typescript
// Returns true if userId has a handle binding that matches either participant
// in the given direct channel (e.g. "_d.alice.bob").
export async function checkDirectChannelAccess(
  vctx: VibesApiSQLCtx,
  channelUserSlug: string,
  userId: string
): Promise<Result<boolean>> {
  const participants = directChannelParticipants(channelUserSlug);
  if (!participants) return Result.Ok(false);
  const [userHandleA, userHandleB] = participants;
  const t = vctx.sql.tables.userHandleBinding;
  const matches = await vctx.sql.db
    .select({ userHandle: t.userHandle })
    .from(t)
    .where(and(eq(t.userId, userId), inArray(t.userHandle, [userHandleA, userHandleB])));
  return Result.Ok(matches.length > 0);
}
```

```typescript
export const listDmThreadsEvento: EventoHandler<W3CWebSocketEvent, MsgBase<ReqListDmThreads>, ResListDmThreads | VibesDiyError> = {
  hash: "list-dm-threads",
  validate: unwrapMsgBase(async (msg: MsgBase) => {
    const ret = reqListDmThreads(msg.payload);
    if (ret instanceof type.errors) return Result.Ok(Option.None());
    return Result.Ok(Option.Some({ ...msg, payload: ret }));
  }),
  handle: checkAuth(
    async (
      ctx: HandleTriggerCtx<W3CWebSocketEvent, MsgBase<ReqWithVerifiedAuth<ReqListDmThreads>>, ResListDmThreads | VibesDiyError>
    ): Promise<Result<EventoResultType>> => {
      const req = ctx.validated.payload;
      const vctx = ctx.ctx.getOrThrow<VibesApiSQLCtx>("vibesApiCtx");
      const userId = req._auth.verifiedAuth.claims.userId;

      // Get all handles for this user
      const t_usb = vctx.sql.tables.userHandleBinding;
      const handleRows = await vctx.sql.db.select({ userHandle: t_usb.userHandle }).from(t_usb).where(eq(t_usb.userId, userId));
      const myUserSlugs = handleRows.map((r) => r.userHandle);

      if (myUserSlugs.length === 0) {
        await ctx.send.send(ctx, { type: "vibes.diy.res-list-dm-threads", status: "ok", items: [] } satisfies ResListDmThreads);
        return Result.Ok(EventoResult.Continue);
      }

      // Get all channel handles where any of my handles participates
      const t_idx = vctx.sql.tables.directChannelIndex;
      const channelRows = await vctx.sql.db
        .select({ channelUserSlug: t_idx.channelUserSlug, userHandle: t_idx.userHandle })
        .from(t_idx)
        .where(inArray(t_idx.userHandle, myUserSlugs));

      const t_docs = vctx.sql.tables.appDocuments;
      const t_reads = vctx.sql.tables.directChannelReads;
      const limit = req.pager?.limit ?? 50;

      const items: DmThreadItem[] = await Promise.all(
        channelRows.map(async ({ channelUserSlug, userHandle: myUserSlug }) => {
          const otherUserSlug = (directChannelParticipants(channelUserSlug) ?? []).find((h) => h !== myUserSlug) ?? "";

          const latestDoc = await vctx.sql.db
            .select()
            .from(t_docs)
            .where(
              and(
                eq(t_docs.userHandle, channelUserSlug),
                eq(t_docs.appSlug, "dm"),
                eq(t_docs.dbName, "messages"),
                eq(t_docs.deleted, 0)
              )
            )
            .orderBy(desc(t_docs.seq))
            .limit(1)
            .then((r) => r[0]);

          const readRow = await vctx.sql.db
            .select({ lastSeenSeq: t_reads.lastSeenSeq })
            .from(t_reads)
            .where(and(eq(t_reads.channelUserSlug, channelUserSlug), eq(t_reads.userHandle, myUserSlug)))
            .then((r) => r[0]);

          const latestSeq = latestDoc?.seq ?? 0;
          const lastSeen = readRow?.lastSeenSeq ?? 0;
          const unreadCount = Math.max(0, latestSeq - lastSeen);

          return {
            channelUserSlug,
            otherUserSlug,
            latestSeq,
            unreadCount,
            latestMessage: latestDoc
              ? {
                  body: String((latestDoc.data as { body?: unknown }).body ?? ""),
                  createdAt: latestDoc.created,
                  authorUserSlug: String((latestDoc.data as { authorUserSlug?: unknown }).authorUserSlug ?? ""),
                }
              : undefined,
          };
        })
      );

      const sorted = items
        .sort((a, b) => ((b.latestMessage?.createdAt ?? "") > (a.latestMessage?.createdAt ?? "") ? 1 : -1))
        .slice(0, limit);

      await ctx.send.send(ctx, { type: "vibes.diy.res-list-dm-threads", status: "ok", items: sorted } satisfies ResListDmThreads);
```

```typescript
export const markDmReadEvento: EventoHandler<W3CWebSocketEvent, MsgBase<ReqMarkDmRead>, ResMarkDmRead | VibesDiyError> = {
  hash: "mark-dm-read",
  validate: unwrapMsgBase(async (msg: MsgBase) => {
    const ret = reqMarkDmRead(msg.payload);
    if (ret instanceof type.errors) return Result.Ok(Option.None());
    return Result.Ok(Option.Some({ ...msg, payload: ret }));
  }),
  handle: checkAuth(
    async (
      ctx: HandleTriggerCtx<W3CWebSocketEvent, MsgBase<ReqWithVerifiedAuth<ReqMarkDmRead>>, ResMarkDmRead | VibesDiyError>
    ): Promise<Result<EventoResultType>> => {
      const req = ctx.validated.payload;
      const vctx = ctx.ctx.getOrThrow<VibesApiSQLCtx>("vibesApiCtx");
      const userId = req._auth.verifiedAuth.claims.userId;

      // Verify this user is a participant
      const rAccess = await checkDirectChannelAccess(vctx, req.channelUserSlug, userId);
      if (rAccess.isErr() || !rAccess.Ok()) {
        await ctx.send.send(ctx, { type: "vibes.diy.res-error", error: { message: "Access denied" } } satisfies ResError);
        return Result.Ok(EventoResult.Continue);
      }

      // Resolve authed handle
      const participants = directChannelParticipants(req.channelUserSlug) ?? ["", ""];
      const t_usb = vctx.sql.tables.userHandleBinding;
      const handleRow = await vctx.sql.db
        .select({ userHandle: t_usb.userHandle })
        .from(t_usb)
        .where(and(eq(t_usb.userId, userId), inArray(t_usb.userHandle, participants)))
        .then((r) => r[0]);
      const myUserSlug = handleRow?.userHandle ?? "";

      const t_reads = vctx.sql.tables.directChannelReads;
      await vctx.sql.db
        .insert(t_reads)
        .values({ channelUserSlug: req.channelUserSlug, userHandle: myUserSlug, lastSeenSeq: req.lastSeenSeq })
        .onConflictDoUpdate({
          target: [t_reads.channelUserSlug, t_reads.userHandle],
          // Use MAX so a delayed/duplicate call never regresses the watermark
          set: { lastSeenSeq: sql`MAX(${t_reads.lastSeenSeq}, ${req.lastSeenSeq})` },
        });

      await ctx.send.send(ctx, { type: "vibes.diy.res-mark-dm-read", status: "ok" } satisfies ResMarkDmRead);
      return Result.Ok(EventoResult.Continue);
    }
  ),
};
```

```typescript
// vibes.diy/pkg/app/components/DmThread.tsx
import React, { useEffect, useRef, useState } from "react";
import { useVctx } from "../contexts/vctx-context.js";
import { directChannelUserSlug } from "@vibes.diy/api-types";

interface DmThreadProps {
  myUserSlug: string;
  otherUserSlug: string;
  vibeRef?: { userHandle: string; appSlug: string };
}

interface MessageDoc {
  _id: string;
  seq: number;
  body?: string;
  authorUserSlug?: string;
  createdAt?: string;
}

export function DmThread({ myUserSlug, otherUserSlug, vibeRef }: DmThreadProps) {
  const vctx = useVctx();
  const channelUserSlug = directChannelUserSlug(myUserSlug, otherUserSlug);
  const [messages, setMessages] = useState<MessageDoc[]>([]);
  const [body, setBody] = useState("");
  const [sending, setSending] = useState(false);
  const bottomRef = useRef<HTMLDivElement>(null);

  async function loadMessages() {
    if (!vctx.vibeDiyApi) return;
    const res = await vctx.vibeDiyApi.queryDocs({
      userHandle: channelUserSlug,
      appSlug: "dm",
      dbName: "messages",
    });
    if (res.isErr()) return;
    setMessages(
      res.Ok().docs.map((d) => ({
        _id: d._id,
        seq: d._seq,
        ...(d as unknown as { body?: string; authorUserSlug?: string; createdAt?: string }),
      }))
    );
  }

  useEffect(() => {
    void loadMessages();
    if (!vctx.vibeDiyApi) return;
    void vctx.vibeDiyApi.subscribeDocs({
      userHandle: channelUserSlug,
      appSlug: "dm",
      dbName: "messages",
    });
    const unsubscribe = vctx.vibeDiyApi.onDocChanged((us, as, db) => {
      if (us === channelUserSlug && as === "dm" && db === "messages") {
        void loadMessages();
      }
    });
    return unsubscribe;
  }, [vctx.vibeDiyApi, channelUserSlug]);

  // Mark as read when messages change
  useEffect(() => {
    if (!messages.length || !vctx.vibeDiyApi) return;
    const latestSeq = Math.max(...messages.map((m) => m.seq));
    void vctx.vibeDiyApi.markDmRead({ channelUserSlug, lastSeenSeq: latestSeq });
  }, [messages, channelUserSlug, vctx.vibeDiyApi]);

  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages]);

  async function handleSend() {
    if (!body.trim() || !vctx.vibeDiyApi) return;
    setSending(true);
    await vctx.vibeDiyApi.putDoc({
      userHandle: channelUserSlug,
      appSlug: "dm",
      dbName: "messages",
      doc: {
        body: body.trim(),
        authorUserSlug: myUserSlug,
        createdAt: new Date().toISOString(),
        ...(vibeRef ? { vibeRef } : {}),
      },
    });
    setBody("");
    setSending(false);
  }

  return (
    <div className="flex flex-col h-full">
      <div className="flex-1 overflow-y-auto p-4 space-y-3">
        {messages.map((m) => (
          <div
            key={m._id}
            className={`flex flex-col ${m.authorUserSlug === myUserSlug ? "items-end" : "items-start"}`}
          >
            <div
              className={`max-w-xs px-3 py-2 rounded-2xl text-sm ${
                m.authorUserSlug === myUserSlug
                  ? "bg-blue-500 text-white"
                  : "bg-gray-100 dark:bg-gray-800 text-gray-900 dark:text-gray-100"
              }`}
            >
              {m.body}
            </div>
            <span className="text-xs text-gray-400 mt-1">
              {m.createdAt ? new Date(m.createdAt).toLocaleTimeString([], { hour: "2-digit", minute: "2-digit" }) : ""}
            </span>
          </div>
        ))}
        <div ref={bottomRef} />
      </div>
      <div className="border-t p-3 flex gap-2">
        <input
          className="flex-1 rounded-full border px-4 py-2 text-sm outline-none focus:ring-2 focus:ring-blue-500"
          placeholder={`Message ${otherUserSlug}…`}
          value={body}
          onChange={(e) => setBody(e.target.value)}
          onKeyDown={(e) => { if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); void handleSend(); } }}
          disabled={sending}
        />
        <button
          onClick={() => void handleSend()}
          disabled={!body.trim() || sending}
          className="rounded-full px-4 py-2 text-sm font-medium bg-blue-500 text-white disabled:opacity-40"
        >
          Send
        </button>
      </div>
    </div>
  );
}
```

```typescript
import { DmThread } from "../components/DmThread.js";

export default function MessageThreadRoute() {

            from {vibeRef.userHandle}/{vibeRef.appSlug}
          </span>
        )}
      </div>
      <DmThread myUserSlug={myUserSlug} otherUserSlug={otherUserSlug} vibeRef={vibeRef} />
    </div>
```
